\documentclass[
  book,                 % gives \chapter, \frontmatter/\mainmatter/\backmatter,
                         % twoside-by-default layout — jlreq has no separate
                         % jlreqbook.cls, book is a class *option*
  paper=a5,              % trim size. Other common choices:
                         %   paper=b6            (~128x182mm, novel/bunko-ish)
                         %   paper={127mm,188mm} (JIS shinsho size, custom dims)
  fontsize=10.5pt,       % Latin body size
  jafontsize=10.5pt,     % Japanese body size (set both explicitly so
                         % jafontscale is not silently applied)
  twoside,
  openright,             % chapters start on a right (recto) page — standard
                         % for books; use openany for report-style docs
  hanging_punctuation,   % allows 、。」 etc. to hang past the line edge
                         % instead of forcing a justification adjustment
  open_bracket_pos=zenkakunibu_nibu,
                         % opening brackets「『(: full-width at a paragraph
                         % start, half-width at a mid-paragraph line break —
                         % this is the common convention in printed books
]{jlreq}

% ----------------------------------------------------------------------
% Fonts
% ----------------------------------------------------------------------
% jlreq deliberately has no font-selection mechanism of its own.
% On LuaLaTeX you bring in fontspec's Japanese-aware counterpart:
\usepackage[match]{luatexja-fontspec}
% "match" makes \bfseries/\itshape switch the CURRENT Japanese font's
% weight/shape too when they're triggered from Latin text, instead of
% only affecting the Latin half of mixed-script text.

% --- Latin (European) fonts -------------------------------------------
\setmainfont{Latin Modern Roman}
\setsansfont{Latin Modern Sans}
\setmonofont{Latin Modern Mono}
% Swap these for whatever you actually want the Latin text set in —
% they only affect the ASCII/Latin portions of the text, not kanji/kana.

% --- Japanese fonts ------------------------------------------------
% Below uses Noto CJK, which is freely available, has real weights
% (unlike most free Mincho faces, which only ship one weight), and is
% installed via `fonts-noto-cjk` on Debian/Ubuntu.
%\setmainjfont[BoldFont=Noto Serif CJK JP SemiBold]{Noto Serif CJK JP}
\setmainjfont[BoldFont=Noto Sans CJK JP Medium]{Noto Serif CJK JP}
\setsansjfont{Noto Sans CJK JP}
% BoldFont matters: most Japanese "Mincho" families ship only a Regular
% weight, so \textbf{...} / \bfseries in Japanese text silently does
% nothing (or, in classic (u)pLaTeX-style setups, jlreq falls back to
% substituting the Gothic face for "bold"). Noto Serif CJK actually has
% distinct weight instances (ExtraLight/Light/Regular/Medium/SemiBold/
% Bold/Black), so pointing BoldFont at one of those gives you a real
% bold Mincho instead of a font-shape hack.
%
% Other solid free options, same syntax:
%   \setmainjfont{Harano Aji Mincho}     — what TeX Live embeds by
%   \setsansjfont{Harano Aji Gothic}       default if you skip this
%                                          section entirely; no bold weight
%   \setmainjfont[BoldFont=BIZ UDPMincho Bold]{BIZ UDPMincho}
%   \setsansjfont[BoldFont=BIZ UDPGothic Bold]{BIZ UDPGothic}
%                                        — Morisawa's UD faces, 2 weights
%                                          each, very legible at body size
%
% If you'd rather not hand-pick weights, luatexja-preset gives you a
% ready-made bundle instead of the block above:
%   \usepackage[haranoaji]{luatexja-preset}   % or: noto-otc / noto / ipa

% ----------------------------------------------------------------------
% Quotation and footnote design
% ----------------------------------------------------------------------
\jlreqsetup{
  % -- block quotations (the `quote` / `quotation` environments) --
  quote_indent           = 2\zw,        % indent from both margins
  quote_end_indent       = 0\zw,
  quote_beforeafter_space = .5\baselineskip,
  quote_fontsize         = small,       % normalsize/small/footnotesize/scriptsize/tiny
  % -- footnotes --
  footnote_second_indent = 1\zw,        % indent for wrapped footnote lines
  reference_mark         = inline,      % where the note marker itself sits;
                                         % 'interlinear' puts it above the
                                         % line instead (more traditional,
                                         % but crowds narrow measures)
  % -- appendix chapter numbering: 付録A, 付録B... instead of 第A章 --
  appendix_heading = {
    chapter = {label_format = {付録\thechapter}}
  },
}

\usepackage{xcolor}
\definecolor{accentcolor}{HTML}{5EAF30}

\ModifyHeading{chapter}{format={\color{accentcolor}#1#2}}
\ModifyHeading{section}{format={\color{accentcolor}#1#2}}

\usepackage[
  backend=biber,
  style=numeric,     % [1][2][3]-style — reads cleanly inline in running
                      % text; switch to style=alphabetic for [Tan21]-style
                      % if you'd rather the marker hint at the source
  sorting=none,       % order of first citation, not alphabetical —
                      % matches how a lecture actually references things
]{biblatex}
\addbibresource{refs.bib}   % keep this as its own tracked file in the
                            % Overleaf/GitHub repo, not inlined in main.tex
\setlength{\biblabelsep}{1em}

% Note: 「」『』 kagikakko dialogue/quotation marks need no package or
% setup at all — they're ordinary characters, and jlreq positions them
% (spacing, line-break behavior, hanging at line-end) per the JLReq
% spec automatically. The settings above only govern the `quote`/
% `quotation` *environment* (indented block quotes), not inline 「」.

% Recommended for any real book (not required by jlreq itself):
\usepackage[hidelinks]{hyperref}   % TOC/cross-reference links in the PDF
